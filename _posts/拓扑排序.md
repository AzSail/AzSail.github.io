#include <bits/stdc++.h>
using namespace std;
const int N = 1010;

int n, m;
int e[N], ne[N], h[N], idx;
int d[N], q[N];

void add(int a, int b){
    e[idx] = b, ne[idx] = h[a], h[a] = idx++;
}

bool topsort(){
    int ll = 0, rr = -1;
    
    for (int i = 1; i <= n; i++) 
        if (!d[i]) q[++rr] = i;
    while (ll <= rr) {
        int t = q[ll++];
        
        for (int i = h[t]; i != -1; i = ne[i]) {
            int j = e[i];
            d[j]--;
            if (!d[j]) q[++rr] = j;
        }
    }
    
    return rr == n - 1;
}

int main(){
    cin >> n >> m;
    
    memset(h, -1, sizeof h);
    
    while (m--) {
        int a, b, c;
        cin >> a >> b;
        add(a, b);
        d[b]++;
    }
    
    if (topsort()) cout << "Yes" << endl;
    else cout << "No" << endl;
    
    return 0;
}
